read_zip_head <- function(zip_path, member, sep, n = 50000) {
  data.table::fread(
    cmd = sprintf("unzip -p %s %s", shQuote(zip_path), shQuote(member)),
    sep = sep,
    nrows = n,
    na.strings = c("", "NA", " "),
    colClasses = "character",
    showProgress = FALSE,
    data.table = FALSE
  ) |>
    tibble::as_tibble()
}

normalise_names <- function(x) {
  x |>
    rename_with(~ sub("^V1$", "row_index", .x)) |>
    rename_with(~ gsub('^"|"$', "", .x))
}

read_asma_sample <- function(dataset, member, n = 50000) {
  archive <- dplyr::filter(asma_archives, DATASET == dataset)
  sep <- if (dataset == "annual_2023_2025") ";" else ","

  read_zip_head(archive$ZIP_PATH, member, sep = sep, n = n) |>
    normalise_names() |>
    mutate(
      DATASET = dataset,
      MEMBER = member,
      YEAR = stringr::str_extract(member, "20[0-9]{2}"),
      MONTH_TOKEN = stringr::str_match(member, "kpi08_([a-z]{3})20[0-9]{2}\\.csv$")[, 2],
      .before = 1
    )
}

present <- function(x) {
  !is.na(x) & trimws(as.character(x)) != ""
}

summarise_presence <- function(x) {
  n <- nrow(x)
  tibble::tibble(
    N_SAMPLE = n,
    ARR_LIKE_ADES_SB = sum(grepl("^SB", x$ades), na.rm = TRUE),
    DEP_LIKE_ADEP_SB = sum(grepl("^SB", x$adep), na.rm = TRUE),
    KPI08_POPULATED = sum(present(x$kpi08), na.rm = TRUE),
    TRANSITO_POPULATED = sum(present(x$transito), na.rm = TRUE),
    DESIMP_POPULATED = sum(present(x$desimp), na.rm = TRUE),
    C_TIME_POPULATED = sum(present(x$c_time), na.rm = TRUE),
    BEAR_POPULATED = sum(present(x$bear), na.rm = TRUE),
    AIBT_POPULATED = sum(present(x$aibt), na.rm = TRUE),
    ALDT_POPULATED = sum(present(x$aldt), na.rm = TRUE),
    SIBT_POPULATED = sum(present(x$sibt), na.rm = TRUE)
  ) |>
    mutate(across(-N_SAMPLE, ~ round(.x / N_SAMPLE, 3), .names = "{.col}_SHARE"))
}

parse_duration_minutes <- function(x) {
  x <- trimws(as.character(x))
  x[x == "" | x == "NA"] <- NA_character_
  sign <- ifelse(grepl("^-", x), -1, 1)
  x <- sub("^-", "", x)

  parts <- stringr::str_match(x, "^([0-9]+):([0-9]{2}):([0-9]{2}(?:\\.[0-9]+)?)$")
  out <- suppressWarnings(
    as.numeric(parts[, 2]) * 60 +
      as.numeric(parts[, 3]) +
      as.numeric(parts[, 4]) / 60
  )
  sign * out
}

This note is a first-pass ingestion check for the two new Brazil ASMA archives received on 22 April 2026. It reads the zip files directly from Downloads and samples each source member, without unpacking the large CSVs into the project tree.

The checks below focus on:

  • coverage and file layout;
  • whether APDF-like input fields can be mapped;
  • whether kpi08 and additional ASMA time fields are already calculated;
  • whether the payload is arrival-only or contains usable departure/DSMA evidence.

The files are close to an APDF arrival extract, but not a drop-in APDF source:

  • the basic APDF identity, airport, aircraft, runway, stand, movement, block, and scheduled fields are available or can be mapped;
  • C40_CROSS_TIME/C40_BEARING and C100_CROSS_TIME/C100_BEARING are not explicitly named;
  • the supplied c_time and bear likely represent one ASMA crossing point, but the file itself does not say whether that point is C40, C100, or another ASMA reference;
  • there is no explicit SRC_PHASE; the data appear arrival-oriented because ades is the Brazilian airport and aldt/aibt are arrival timestamps.

The new ASMA files do not look like departure DSMA files because they use arrival fields (aldt, aibt, sibt) and have no explicit departure movement or off-block/take-off timestamps. The presence of Brazilian adep values mostly reflects domestic arrivals where both ends of the pair are in Brazil.

Preliminary Conclusions

  • Both new archives provide calculated kpi08.
  • Both provide additional ASMA time fields: transito and desimp, plus c_time and bear for a crossing-point reference.
  • The main structural difference is annual versus monthly packaging, delimiter, and the monthly archive’s leading row index.
  • The monthly archive needs a duplicate check for feb versus fev.
  • The data are arrival-oriented ASMA data and appear usable for ARR KPI08/APDF-like ingestion.
  • The files do not appear to provide true DEP/DSMA evidence; a separate DSMA source would still be needed unless DECEA confirms that departure fields are encoded elsewhere.

Next Ingestion Steps

  1. Confirm with DECEA what c_time/bear represent: C40, C100, or another ASMA reference.
  2. Confirm whether transito, desimp, and kpi08 definitions match KPI08 calculation expectations.
  3. Decide whether to use the annual 2025 file or reconstruct 2025 from monthly files after resolving feb/fev.
  4. Build a harmonised parquet extract with APDF-like names for ARR only, retaining the original calculated KPI fields for validation.

Production Ingestion

The first production-style ingestion is implemented in R/08-ingest-bra-asma.R. It uses the multi-year annual archive and keeps the raw Brazil CLASS value as supplied. At this stage, CLASS should therefore be interpreted as the Brazil/WTC-style category available in the source, not as the APDF class code that combines wake-turbulence category with propulsion.

The script writes:

  • APDF-like ARR ASMA parquet extracts to /Users/rainerkoelle/RProjects/PBWG-BRA/data/asma;
  • 2024 GANP p20 reference files to /Users/rainerkoelle/RProjects/PBWG-BRA/outputs/asma-reference-2024;
  • daily analytic ASMA files to /Users/rainerkoelle/RProjects/PBWG-BRA/outputs/asma-daily-2023, asma-daily-2024, and asma-daily-2025;
  • combined report-layer outputs to data/.

The reference is calculated from 2024 valid arrival ASMA samples by:

ICAO, PHASE, RANGE, CLASS, RWY, and SECTOR_GROUP.

The daily analytic output keeps the same breakdown fields and reports movement counts plus total observed ASMA time, total reference time, total additional time, and the Brazil-supplied reference/KPI08 totals for comparison.
